import { pgTable, serial, varchar, date, integer, uuid } from "drizzle-orm/pg-core";

export const cars = pgTable("cars", {
  id: serial("id").primaryKey(),
  name: varchar("name", { length: 255 }),
  pricePerDay: integer("price_per_day"),
});

export const bookings = pgTable("bookings", {
  id: serial("id").primaryKey(),
  carId: integer("car_id").references(() => cars.id),
  userId: uuid("user_id"),
  startDate: date("start_date"),
  endDate: date("end_date"), // exclusive
});

// routes/bookings.ts

import { eq, and, or } from "drizzle-orm";
import { db } from "../drizzle/db";
import { bookings } from "../drizzle/schema";
import express from "express";

const router = express.Router();

router.post("/book", async (req, res) => {
  const { carId, userId, startDate, endDate } = req.body;

  // check for overlaps
  const overlaps = await db
    .select()
    .from(bookings)
    .where(
      and(
        eq(bookings.carId, carId),
        or(
          and(
            bookings.startDate.lte(new Date(endDate)),
            bookings.endDate.gt(new Date(startDate))
          )
        )
      )
    );

  if (overlaps.length > 0) {
    return res.status(400).json({ message: "Dates not available" });
  }

  // Insert booking
  await db.insert(bookings).values({
    carId,
    userId,
    startDate: new Date(startDate),
    endDate: new Date(endDate),
  });

  res.status(201).json({ message: "Booking confirmed" });
});

export default router;

// components/BookingCalendar.tsx
import { DateRange } from "react-day-picker";
import { useQuery } from "@tanstack/react-query";
import { addDays, eachDayOfInterval, parseISO } from "date-fns";

type Booking = { startDate: string; endDate: string };

const getBookedDates = (bookings: Booking[]) => {
  return bookings.flatMap(({ startDate, endDate }) =>
    eachDayOfInterval({
      start: parseISO(startDate),
      end: addDays(parseISO(endDate), -1),
    })
  );
};

export default function BookingCalendar({ carId }: { carId: number }) {
  const { data: bookings } = useQuery({
    queryKey: ["bookings", carId],
    queryFn: async () => {
      const res = await fetch(`/api/cars/${carId}/bookings`);
      return res.json();
    },
  });

  const disabledDates = bookings ? getBookedDates(bookings) : [];

  return (
    <div className="p-4">
      <DatePicker
        mode="range"
        numberOfMonths={2}
        disabled={(date) => disabledDates.some((d) => d.toDateString() === date.toDateString())}
        // onSelect={handleBookingSelect}
      />
    </div>
  );
}

router.get("/:carId/bookings", async (req, res) => {
  const carId = parseInt(req.params.carId);

    .select({
      startDate: bookings.startDate,
      endDate: bookings.endDate,
    })
    .from(bookings)
    .where(eq(bookings.carId, carId));

  res.json(carBookings);
